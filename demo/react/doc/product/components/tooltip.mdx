() => {

                Right Tooltip
            </Tooltip>
            <Tooltip anchorRef={anchorRefBottom} placement="bottom">

            </Tooltip>
            <Tooltip anchorRef={anchorRefLeft} placement="left">
                Left Tooltip
            </Tooltip>
            <Tooltip anchorRef={anchorRefDelay} delay={2000} placement="bottom">
                Tooltip with delay
            </Tooltip>
        </>
    );
};
