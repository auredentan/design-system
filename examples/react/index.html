This is an example of using LumX with React

You have multiple choices to use LumX with React:

  • Browser: All you have to do is import lumx-theme-<lumapps|material>[.min].css in your page's head and lumx.react[.min].js at the end of your page's body after React and ReactDOM library. 

                            <!DOCTYPE html>
                        <html ng-app="lumx-example-app">
                            <head>
                                <title>React browser example</title>
                                <link rel="stylesheet" href="node_modules/@lumx/dist/lumx-theme-lumapps.min.css" />
                            </head>

                            <body class="p++">
                                <div id="root"></div>

                                <script
                                    src="https://unpkg.com/@babel/standalone/babel.min.js"
                                    type="text/javascript"
                                ></script>
                                <script
                                    src="https://unpkg.com/react/umd/react.production.min.js"
                                    type="text/javascript"
                                ></script>
                                <script
                                    src="https://unpkg.com/react-dom/umd/react-dom.production.min.js"
                                    type="text/javascript"
                                ></script>
                                <script src="node_modules/@lumx/dist/lumx.react.min.js" type="text/javascript"></script>

                                <script type="text/babel">
                                    const { Fragment, PureComponent } = React;
                                    const { LxButton } = LumX;

                                    class MyButton extends PureComponent {
                                        render() {
                                            return (
                                                <LxButton color={this.props.color}>{this.props.children}</LxButton>
                                            );
                                        }
                                    }

                                    const MyFunctionnalButton = (props) => {
                                        return (
                                            <LxButton color={props.color}>{props.children}</LxButton>
                                        );
                                    };

                                    ReactDOM.render(
                                        <Fragment>
                                        <div class="lumx-typography-title">This is a basic example</div>
                                        <MyButton>Nice!</MyButton>
                                        <MyFunctionnalButton color="green">Even nicer!</MyFunctionnalButton>
                                        </Fragment>,
                                        document.getElementById('root')
                                    );
                                </script>
                            </body>
                        </html>

  • CommonJS[2]: All you have to do is import lumx-theme-<lumapps|material>[.min].css in your page's head and the components you want to use in your own React components. All components comes packaged with their own CSS, so you don't need to import any additionnal CSS in your application. 

                            <!DOCTYPE html>
                        <html ng-app="lumx-example-app">
                            <head>
                                <title>React CommonJS example</title>
                                <link rel="stylesheet" href="node_modules/@lumx/dist/lumx-theme-lumapps.min.css" />
                            </head>

                            <body class="p++">
                                <div id="root"></div>

                                <script
                                    src="https://unpkg.com/@babel/standalone/babel.min.js"
                                    type="text/javascript"
                                ></script>
                                <script
                                    src="https://unpkg.com/react/umd/react.production.min.js"
                                    type="text/javascript"
                                ></script>
                                <script
                                    src="https://unpkg.com/react-dom/umd/react-dom.production.min.js"
                                    type="text/javascript"
                                ></script>
                                <script src="node_modules/@lumx/dist/lumx.react.min.js" type="text/javascript"></script>

                                <script type="text/javascript">
                                    const { Fragment, PureComponent } = React;
                                    const { LxButton } = require('@lumx/react');

                                    class MyButton extends PureComponent {
                                        render() {
                                            return (
                                                <LxButton color={this.props.color}>{this.props.children}</LxButton>
                                            );
                                        }
                                    }

                                    const MyFunctionnalButton = (props) => {
                                        return (
                                            <LxButton color={props.color}>{props.children}</LxButton>

                                        </Fragment>,
                                        document.getElementById('root')
                                    );
                                </script>
                            </body>
                        </html>

  • AMD: All you have to do is import lumx-theme-<lumapps|material>[.min].css in your page's head and the components you want to use in your own React components. All components comes packaged with their own CSS, so you don't need to import any additionnal CSS in your application. 

                            <!DOCTYPE html>
                        <html ng-app="lumx-example-app">
                            <head>
                                <title>React AMD example</title>
                                <link rel="stylesheet" href="node_modules/@lumx/dist/lumx-theme-lumapps.min.css" />
                            </head>

                            <body class="p++">
                                <div id="root"></div>

                                <script
                                    src="https://unpkg.com/@babel/standalone/babel.min.js"
                                    type="text/javascript"
                                ></script>
                                <script
                                    src="https://unpkg.com/react/umd/react.production.min.js"
                                    type="text/javascript"
                                ></script>
                                <script
                                    src="https://unpkg.com/react-dom/umd/react-dom.production.min.js"
                                    type="text/javascript"
                                ></script>
                                <script src="node_modules/@lumx/dist/lumx.react.min.js" type="text/javascript"></script>

                                <script type="text/javascript">
                                    require(['@lumx/react'], (LumX) => {
                                        const { Fragment, PureComponent } = React;
                                        const { LxButton } = LumX;

                                        class MyButton extends PureComponent {
                                            render() {
                                                return (
                                                    <LxButton color={this.props.color}>{this.props.children}</LxButton>
                                                );
                                            }
                                        }

                                        const MyFunctionnalButton = (props) => {
                                            return (
                                                <LxButton color={props.color}>{props.children}</LxButton>
                                            );
                                        };

                                        ReactDOM.render(
                                            <Fragment>
                                            <div class="lumx-typography-title">This is a basic example</div>
                                            <MyButton>Nice!</MyButton>
                                            <MyFunctionnalButton color="green">Even nicer!</MyFunctionnalButton>
                                            </Fragment>,
                                            document.getElementById('root')
                                        );
                                    });
                                </script>
                            </body>
                        </html>
